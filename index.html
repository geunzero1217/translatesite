const API_KEY = "여기에_당신의_OpenAI_API_KEY_입력"; // ⚠️ 실제 배포 시 노출 금지 function startTranslation() { const recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)(); recognition.lang = 'en-US'; recognition.continuous = true; recognition.interimResults = false; recognition.onresult = async function(event) { const englishText = event.results[event.results.length - 1][0].transcript; console.log("🎧 영어 인식:", englishText); // 번역 요청 const koreanText = await translateToKorean(englishText); document.getElementById("subtitle").innerText = koreanText; }; recognition.onerror = function(event) { alert("음성 인식 오류: " + event.error); }; recognition.start(); } async function translateToKorean(text) { const response = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${API_KEY}`, }, body: JSON.stringify({ model: "gpt-4", messages: [ { role: "system", content: "Translate English to Korean." }, { role: "user", content: text } ] }) }); const data = await response.json(); return data.choices[0].message.content.trim(); }
